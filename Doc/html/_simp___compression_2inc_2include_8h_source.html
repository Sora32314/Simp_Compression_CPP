186 bool Gene_Compression(std::string&& argPath, bool isRelativeArg, std::string&& resPath, bool isRelativeRes)
187 {
188 std::vector<int> streamTarget;
189 //int位大小
190 short intBitCount = sizeof(int) * 8;
191
192 int target = 0;
193 fs::path fsResPath(std::forward<std::string>(resPath));
194 fs::path fsArgPath(std::forward<std::string>(argPath));
195 //如果是相对路径，则补充为绝对路径
196 if(isRelativeRes)
197 {
198 fsResPath = fs::current_path() / fsResPath;
199 }
200
201 if(isRelativeArg)
202 {
203 fsArgPath = fs::current_path() / fsArgPath;
204 }
205
206 //打开文件
207 std::ifstream argFile(fsArgPath, std::ios::in);
208 std::ofstream resFile(fsResPath, std::ios::out|std::ios::trunc|std::ios::binary);
209
210 //文件长度
211 size_t length = 0;
212 resFile.write(reinterpret_cast<char*>(&length), sizeof(size_t));
213
214 //逐渐读取文件直到文件末尾
215 char buffer[128] = {0};
216 while (!argFile.eof())
217 {
218 //创建缓冲区并初始化字符串
219 argFile.read(buffer, 128);
220 std::streamsize bytesRead = argFile.gcount();
221
222 //将读取到的数据存储至数组中
223 for(decltype(bytesRead) i = 0; i < bytesRead; i++, length += 2)
224 {
225 if(buffer[i] == 'A')
226 {
227 streamTarget.push_back(0);
228 streamTarget.push_back(0);
229 }
230 else if(buffer[i] == 'T')
231 {
232 streamTarget.push_back(1);
233 streamTarget.push_back(0);
234 }
235 else if(buffer[i] == 'C')
236 {
237 streamTarget.push_back(0);
238 streamTarget.push_back(1);
239 }
240 else if(buffer[i] == 'G')
241 {
242 streamTarget.push_back(1);
243 streamTarget.push_back(1);
244 }
245 }
246
247 //处理并写入数据到int中，之后写入到文件里
248 for (size_t i = 0; i < streamTarget.size();)
249 {
250 target = 0;
251 for (int j = 1; j <= intBitCount && i < streamTarget.size(); j++, i++)
252 {
253 target = BO::Bit_Set(target, j, streamTarget[i]);
254 }
255 resFile.write(reinterpret_cast<char*>(&target), sizeof(int));
256 }
257
258 streamTarget.clear();
259 }
260
261 resFile.seekp(std::ios::beg);
262 resFile.write(reinterpret_cast<char*>(&length), sizeof(size_t));
263
264 argFile.close();
265 resFile.close();
266
267 return true;
268 }

281 bool Gene_Decompression(std::string&& argPath, bool isRelativeArg, std::string&& resPath, bool isRelativeRes)
282 {
283 std::vector<bool> streamTarget;
284 std::vector<char> streamInput;
285 fs::path fsResPath(std::forward<std::string>(resPath));
286 fs::path fsArgPath(std::forward<std::string>(argPath));
287 //如果是相对路径，则补充为绝对路径
288 if(isRelativeRes)
289 {
290 fsResPath = fs::current_path() / fsResPath;
291 }
292
293 if(isRelativeArg)
294 {
295 fsArgPath = fs::current_path() / fsArgPath;
296 }
297
298 std::ifstream ArgFile(fsArgPath, std::ios::in|std::ios::binary);
299 std::ofstream ResFile(fsResPath, std::ios::ate|std::ios::trunc);
300
301
302 //获取文件长度
303 char lengthBuffer[sizeof(size_t)] = {0};
304 ArgFile.read(lengthBuffer, sizeof(size_t));
305 size_t size = *(reinterpret_cast<size_t*>(lengthBuffer));
306
307
308 while(size > 0)
309 {
310 //创建并初始化字符串
311 char buffer[128] = {0};
312 ArgFile.read(buffer, 128);
313 std::streamsize bytesRead = ArgFile.gcount();
314
315 //读取每一位到bool类型数组中
316 for(decltype(bytesRead) i = 0; i < bytesRead; i++)
317 {
318 for(decltype(bytesRead) j = 0; j < BYTESIZE; j++)
319 {
320 streamTarget.push_back(BO::Bit_Check(buffer[i], j));
321 }
322 }
323
324
325
326 for(size_t bufferCount = 0; bufferCount < streamTarget.size() && size > 0; bufferCount += 2, size -= 2)
327 {
328 if(streamTarget[bufferCount] == false)
329 {
330 if(streamTarget[bufferCount + 1] == false)
331 {
332 streamInput.push_back('A');
333 }
334 else
335 {
336 streamInput.push_back('C');
337 }
338 }
339 else if(streamTarget[bufferCount] == true)
340 {
341 if(streamTarget[bufferCount + 1] == false)
342 {
343 streamInput.push_back('T');
344 }
345 else
346 {
347 streamInput.push_back('G');
348 }
349 }
350 }
351
352 streamTarget.clear();
353
354 ResFile.write(streamInput.data(), streamInput.size());
355
356 streamInput.clear();
357 }
358
359 ArgFile.close();
360 ResFile.close();
361
362 return true;
363 }
